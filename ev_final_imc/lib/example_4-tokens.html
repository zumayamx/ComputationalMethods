#include <iostream>
using namespace std;

class Animal {
public:
    virtual void sound() const {
        cout << "Some generic animal sound" << endl;
    }
};

class Dog : public Animal {
public:
    void sound() const override {
        cout << "Woof" << endl;
    }
};

int main() {
    Animal* animal = new Dog();
    animal->sound();
    delete animal;
    return 0;
}
