#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Example {
public:
    Example() : number(0) {}

    void setNumber(int num) {
        number = num;
    }

    int getNumber() const {
        return number;
    }

    void displayMessage() const {
        cout << "The number is: " << number << endl;
    }

private:
    int number;
};

int main() {
    Example example;
    example.setNumber(42);
    example.displayMessage();

    // A simple loop
    for (int i = 0; i < 10; ++i) {
        cout << "i is: " << i << endl;
    }

    // Vector usage
    vector<string> words = {"Hello", "world", "C++", "lexer"};
    for (const auto& word : words) {
        cout << word << " ";
    }
    cout << endl;

    // Conditional statement
    if (example.getNumber() > 0) {
        cout << "Number is positive" << endl;
    } else {
        cout << "Number is non-positive" << endl;
    }

    return 0;
}
