#include <iostream>
#include <string>
using namespace std;

int main() {
    string str = "Hello, world!";
    str.replace(7, 5, "C++");
    cout << str << endl;

    size_t pos = str.find("C++");
    if (pos != string::npos) {
        cout << "Found 'C++' at position " << pos << endl;
    }

    return 0;
}
